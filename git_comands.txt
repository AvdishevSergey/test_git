git status - состояние репозитория
git init - создание локального репозитория на своём компьютере
cd - указать папку в репозитории которую обрабатываем
git add <назване файла> добафить файл в репозиторий 
git config user.email <почта @email.domen> - указать почту пользователя
git config user.name <"Имя Фамилия латиницей"> - указать имя пользователя
git commit <-m> <"что изменилось"> - утвердить изменения в репозитории
git log - история комитов(изменений/версий файлов) в репозитории
